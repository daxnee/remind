### 아이템 길이 체크하기
```js
  function createItem(){
        var len =  $(".row").children().length; 
        // row의 자식 태그가 몇 개인지
        if(len < 6){
            for (var i =0; i<5; i++){
                // 아이템이 배열이기 때문에 0번째 부터 시작하니까 i<5;
             $('.row').append(' <div class="item"></div>');
            }
        }else{
          alert("아이템 길이가 6을 넘었습니다")
        
        }
       }
```
